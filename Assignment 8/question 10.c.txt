#include <stdio.h>

int main(void) {
	/*
	    Pattern 9
	*/
	int k=1;
	
	for(int i=1 ; i<=4 ; i++)
	{
	    k=1;
	    for(int j=1 ;j<=7 ; j++)
	    {
	        
	        if(j<=4-(i-1) || j>=4+(i-1))
	        {
	            printf("%d",k);
	        }
	        else
	        {
	            printf(" ");
	        }
	        (j<4) ? k++ : k--;
	    }
	 
	    printf("\n");
	
	}
	
	return 0;
	
}
